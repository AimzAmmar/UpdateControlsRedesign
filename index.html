Q.E.D. Framework

MVVM with formulas, not events

Install Package

All MVVM frameworks fire PropertyChanged events when you set a property. But what about the other properties affected by the change? Do you have calls to RaisePropertyChaged in your code? You don't need to.

The Q.E.D. Framework discovers dependencies within your code automatically. Just write a formula for each property. It will raise property changed events for you when the property is affected.

Typical MVVM

public class PersonViewModel :
  ViewModelBase
{
  private string _first;
  private string _last;

  public string First
  {
    get { return _first; }
    set 
    {
      _first = value;
      RaisePropertyChanged(() => First);
      RaisePropertyChanged(() => Full);
    }
  }

  public string Last
  {
    get { return _last; }
    set 
    {
      _last = value;
      RaisePropertyChanged(() => Last);
      RaisePropertyChanged(() => Full);
    }
  }

  public string Full
  {
    get
    {
      return First + " " + Last;
    }
  }
}

Q.E.D.

public class PersonViewModel
{
  private Variable<string> _first =
    new Variable<string>();
  private Variable<string> _last =
    new Variable<string>();

  public string First
  {
    get { return _first; }
    set { _first.Value = value; }
  }

  public string Last
  {
    get { return _last; }
    set { _last.Value = value; }
  }

  public string Full
  {
    get
    {
      return First + " " + Last;
    }
  }
}

The Result

The Q.E.D. Framework will even discover dependencies across object boundaries. You don't need to subscribe to events.

Typical MVVM

public class PersonModel : ObservableBase
{
  private string _first;
  private string _last;

  public string First
  {
    get { return _first; }
    set
    {
      _first = value;
      RaisePropertyChanged(() => First);
    }
  }

  public string Last
  {
    get { return _last; }
    set
    {
      _last = value;
      RaisePropertyChanged(() => Last);
    }
  }
}

public class PersonViewModel : ViewModelBase
{
  private readonly PersonModel _person;

  public PersonViewModel(PersonModel person)
  {
    _person = person;
    _person.PropertyChanged += PersonPropertyChanged;
  }

  void PersonPropertyChanged(
    object sender, PropertyChangedEventArgs e)
  {
    if (e.PropertyName == "First")
    {
      RaisePropertyChanged(() => First);
      RaisePropertyChanged(() => Full);
    }
    else if (e.PropertyName == "Last")
    {
      RaisePropertyChanged(() => Last);
      RaisePropertyChanged(() => Full);
    }
  }

  public string First
  {
    get { return _person.First; }
    set { _person.First = value; }
  }

  public string Last
  {
    get { return _person.Last; }
    set { _person.Last = value; }
  }

  public string Full
  {
    get
    {
      return _person.First + " " + _person.Last;
    }
  }
}
